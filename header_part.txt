import { useEffect, useState } from 'react';
import { collection, onSnapshot, orderBy, query, doc, getDoc, setDoc, serverTimestamp } from 'firebase/firestore';
import { db } from '../lib/firebase';
import Plot from 'react-plotly.js';

export default function LivingNewsroom() {
    const [episodes, setEpisodes] = useState<any[]>([]);
    const [primitives, setPrimitives] = useState<any>({});
    const [metrics, setMetrics] = useState<any>({});
    const [selectedEpisode, setSelectedEpisode] = useState<number>(1);

            if (doc.exists()) {
                setPrimitives(doc.data().primitives || {});
            }
        });

            if (doc.exists()) {
                setMetrics(doc.data());
            }
        });

        return () => {
            unsubscribeEpisodes();
            unsubscribePrimitives();
            unsubscribeMetrics();
